+def error_response(message, status_code, details=None):
+    """Return a consistent JSON error response."""
+    payload = {"success": False, "error": message, "status": status_code}
+    if details is not None:
+        payload["details"] = details
+    return jsonify(payload), status_code
+
+
+def handle_exception(exc, context=None):
+    """Map common exception types to HTTP status codes.
+
+    Returns a tuple (json_response, status_code).
+    """
+    message = str(exc) or "An unexpected error occurred"
+    if isinstance(exc, (KeyError, AttributeError)):
+        return error_response("Resource not found", 404, message)
+    if isinstance(exc, ValueError):
+        return error_response("Bad request", 400, message)
+    if isinstance(exc, PermissionError):
+        return error_response("Forbidden", 403, message)
+    if isinstance(exc, (ConnectionError, TimeoutError)):
+        return error_response("Service unavailable", 503, message)
+    if isinstance(exc, NotImplementedError):
+        return error_response("Not implemented", 501, message)
+    return error_response("Internal server error", 500, message if context else None)
